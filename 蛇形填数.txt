#include <iostream>
#include<cstring>
using namespace std;
int a[15][15];
int main()
{
  int n,x,y,tot;
  tot=0;
  cin>>n;
  memset(a,0,sizeof(a));
  tot=1;
  a[x=0][y=0]=1;
  while(tot<n*n)
  {
      while(x+1<n&&!a[x+1][y])
         a[++x][y]=++tot;
      while(y+1<n&&!a[x][y+1])
         a[x][++y]=++tot;
      while(x-1>=0&&!a[x-1][y])
        a[--x][y]=++tot;
      while(y-1>=0&&!a[x][y-1])
        a[x][--y]=++tot;

  }
  for(x=0;x<n;x++)
   {
       for(y=0;y<n;y++)
        cout<<a[x][y]<<' ';
      cout<<endl;

   }
  return 0;
}
